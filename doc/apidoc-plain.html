      7

      "page_id": 8,

      "ocr_xml_path": "/some/path/output.xml",

      "alt_change_index": null,
      "noisiness_idx": null
    }
  ],
  "postproc_results": [
    {
      "page_id": 9,
      "batch_job_id": 14,
      "pp_ecorr": 0.0,
      "pp_juxta": 0.0,
      "pp_retas": 0.0,
      "pp_health": 0.0,

    }

}

{
  "page_results": {
    "imported": 1
  },
  "postproc_results": {
    "imported": 1
  }
}

Params

  • job_queues : required [ Must be a Hash ]
    • completed : required , nil allowed [ Must be an array of any type ]

      Array of Job Queue IDs that completed

    • failed : required , nil allowed [ Must be an array of any type ]

      Array of Job Queue IDs that failed

  • page_results : optional [ Must be an Array of nested elements ]

    Page results

    • page_id : required [ Must be Integer. ]
    • batch_id : required [ Must be Integer. ]
    • ocr_text_path : required [ Must be String ]
    • ocr_xml_path : required [ Must be String ]
    • juxta_change_index : optional , nil allowed [ Must be Float. ]
    • alt_change_index : optional , nil allowed [ Must be Float. ]
    • noisiness_idx : optional , nil allowed [ Must be Float. ]
  • postproc_results : optional [ Must be an Array of nested elements ]

    Post process results

    • page_id : required [ Must be Integer. ]
    • batch_job_id : required [ Must be Integer. ]
    • pp_ecorr : optional , nil allowed [ Must be Float. ]
    • pp_stats : optional , nil allowed [ Must be Float. ]
    • pp_juxta : optional , nil allowed [ Must be Float. ]
    • pp_retas : optional , nil allowed [ Must be Float. ]
    • pp_health : optional , nil allowed [ Must be Float. ]
